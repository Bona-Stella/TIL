import java.util.Stack;

public class Main {
    public int result(String str) {
        Stack<Character> args = new Stack<>();
        for (Character c : str.toCharArray()) {
            if (args.empty()) {
                args.push(c);
                continue;
            }
            if (args.peek() == c) {
                args.pop();
            } else {
                args.push(c);
            }
        }
        return args.isEmpty() ? 1 : 0;
    }

    public static void main(String[] args) {
        Main T = new Main();
        String str = "cdcd";
        System.out.println(T.result(str));
    }
}
