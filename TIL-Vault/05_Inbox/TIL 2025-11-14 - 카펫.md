import java.util.Arrays;

public class Main {
    public int[] result(int brown, int yellow) {
        int total = brown + yellow;

        for (int height = 3; height <= total; height++) {
            for (int width = height; width <= total; width++) {
                if (width * height != total)
                    continue;

                int inner = (width - 2) * (height - 2);
                if (inner == yellow)
                    return new int[] { width, height };
            }
        }
        return new int[] { 0, 0 };
    }

    public static void main(String[] args) {
        Main T = new Main();
        int brown = 10;
        int yellow = 2;
        System.out.println(Arrays.toString(T.result(brown, yellow)));
    }
}
