import java.util.Scanner;
import java.util.Stack;

public class Main {
    public boolean result(String str) {
        boolean output = true;
        Stack<Character> stack = new Stack<>();

        for (char c : str.toCharArray()) {
            if (c == '(') {
                stack.push(c);
            } else {
                if (stack.isEmpty()) {
                    output = false;
                    break;
                }
                stack.pop();
            }
        }

        if (!stack.isEmpty()) {
            output = false;
        }

        return output;
    }

        Scanner scan = new Scanner(System.in);

        System.out.println(T.result(str));
    }
}
