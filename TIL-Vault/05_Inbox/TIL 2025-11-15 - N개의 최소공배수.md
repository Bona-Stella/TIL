public class Main {
    public static int result(int[] arr) {
        Arrays.sort(arr);
        int level = 0;
        boolean flag = true;

        while (flag) {
            level++;
            int temp = level * arr[arr.length - 1];
            for (int i = 0; i < arr.length - 1; i++) {
                if (temp % arr[i] != 0) {
                    flag = true;
                    break;
                } else {
                    flag = false;
                }
            }
        }
        
        return level * arr[arr.length - 1];
    }

    public static void main(String[] args) {
        int[] arr = { 2, 6, 8, 14 };
        System.out.println(result(arr));
    }
}
