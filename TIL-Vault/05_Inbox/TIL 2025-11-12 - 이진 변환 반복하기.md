    public int[] result(String str) {
        int count = 0, level = 0;
        StringBuilder builder = new StringBuilder();

        while (str.length() > 1) {
            for (char c : str.toCharArray()) {
                if (Character.getNumericValue(c) == 0) {
                    count++;
                } else {
                    builder.append(c);
                }
            }
            str = Integer.toBinaryString(builder.length());
            builder.setLength(0);
            level++;
        }

        int[] outputs = new int[2];
        outputs[0] = level;
        outputs[1] = count;
        return outputs;
    }

    public static void main(String[] args) {
        Main T = new Main();
        String str = "110010101001";
        System.out.println(Arrays.toString(T.result(str)));
    }
}
