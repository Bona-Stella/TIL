import java.util.Arrays;

public class Main {
    public int result(int[] people, int limit) {
        Arrays.sort(people);
        int boats = 0;
        int left = 0;
        int right = people.length - 1;

        while (left <= right) {
            boats++;

            if (people[left] + people[right] <= limit) {
                left++;
            }
            
            right--;
        }
        return boats;
    }

    public static void main(String[] args) {
        Main T = new Main();
        int[] people = { 40, 40, 100, 100, 100, 100 };
        int limit = 140;
        System.out.println(T.result(people, limit));
    }
}
